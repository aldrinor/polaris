141:def _impute_cost_from_tokens(
142:    model: str,
143:    input_tokens: int,
144:    output_tokens: int,
145:    reasoning_tokens: int,
146:) -> float:
147:    """Estimate $/call from token counts + published prices. Reasoning
148:    tokens bill at the output rate (matches OpenAI / Anthropic practice).
149:    Used by the budget guard when OpenRouter omits usage.cost.
150:
151:    Defensive: token counts are clamped to >=0. A corrupted API response
152:    returning negative tokens must NOT produce a negative cost that
153:    would silently shrink the accumulated run budget and let a runaway
154:    loop keep calling past PG_MAX_COST_PER_RUN.
155:    """
156:    # Clamp negative token counts — these can only come from a
157:    # corrupted API response. A negative cost here would be silently
158:    # absorbed by _add_run_cost and weaken the budget guard.
159:    input_tokens = max(0, int(input_tokens))
160:    output_tokens = max(0, int(output_tokens))
161:    reasoning_tokens = max(0, int(reasoning_tokens))
162:    if input_tokens == 0 and output_tokens == 0 and reasoning_tokens == 0:
163:        return 0.0
164:    model_lower = (model or "").lower()
165:    input_rate, output_rate = _DEFAULT_PRICE_PER_M
166:    for prefix, (in_r, out_r) in _PRICE_TABLE_USD_PER_M.items():
167:        if model_lower.startswith(prefix):
168:            input_rate, output_rate = in_r, out_r
169:            break
170:    return (
171:        (input_tokens / 1_000_000.0) * input_rate
172:        + ((output_tokens + reasoning_tokens) / 1_000_000.0) * output_rate
173:    )
