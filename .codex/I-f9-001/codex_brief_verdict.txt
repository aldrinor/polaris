function isStale(publication_date: string | null): boolean {
  if (!publication_date) return false;
  const t = new Date(publication_date).getTime();
  if (isNaN(t)) return false;

  token,
  text,
  idx,
  spanIdx,
}: {
  token: ParsedToken;
  text: string;
  idx: number;
  spanIdx: number;
}) {
