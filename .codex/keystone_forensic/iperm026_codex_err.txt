  254: def render_reduce_user(
  255:     distillate: SectionDistillate,
  256: ) -> str:
  257:     """Render the REDUCE user prompt (spec ├В┬з3): the validated findings ledger,
  258:     one line per finding, plus contradiction clusters."""
  259:     ledger_lines: list[str] = []
  260:     for f in distillate.findings:
  261:         atom_ids = ",".join(f.atom_ids)
  262:         quote = f.support_quote.replace("\n", " ")
  263:         ledger_lines.append(
  264:             f"{f.finding_id} | {f.evidence_id} | {f.source_tier} | "
  265:             f"cite=[{f.evidence_id}] | atom_ids={atom_ids} | "
  266:             f"claim={f.claim} | caveat={f.caveat} | "
  267:             f"contradiction_key={f.contradiction_key} | quote={quote}"
  268:         )
  269:     ledger = "\n".join(ledger_lines) if ledger_lines else "(no findings)"
  270: 
  271:     cluster_lines: list[str] = []
  272:     for c in distillate.contradiction_clusters:
  273:         ids = ",".join(c.finding_ids)
  274:         cluster_lines.append(
  275:             f"{c.contradiction_key}: {ids}"
  276:             + (f" ├втВмтАЭ {c.summary}" if c.summary else "")
  277:         )
  278:     clusters = "\n".join(cluster_lines) if cluster_lines else "(none)"
  279: 
  280:     return (
  281:         f"SECTION_TITLE: {distillate.section_title}\n"
  282:         f"SECTION_FOCUS: {distillate.section_focus}\n\n"
  283:         f"VALIDATED_FINDINGS_LEDGER:\n{ledger}\n\n"
  284:         f"CONTRADICTION_CLUSTERS:\n{clusters}\n\n"
  285:         "Write the section now. Each sentence must use ledger facts only and place its "
  286:         "finding marker(s), atom id(s) when needed, and [ev_XXX] marker(s) INSIDE the "
  287:         "sentence, immediately BEFORE the terminal period тАФ never on a separate line or "
  288:         "after the period. Do not write [#ev:...] span tokens."
  289:     )

  406: def _numbers_in_text(text: str) -> set[str]:
  407:     """Numeric tokens under the boundary-aware definition shared with the
  408:     downstream verifier (spec section 1 check 4: "the same numeric definitions
  409:     as strict_verify plus claim_atom_extractor").
  410: 
  411:     Uses claim_atom_extractor._NUMBER_ATOM_RE, whose ``(?<![A-Za-z0-9_.])``
  412:     lookbehind correctly excludes the ``1`` embedded in "HbA1c" (preceded by a
  413:     letter) while still catching ``-1.86``, ``99.9``, and ranges. The bare

  415:     junk ``1`` that would both over-strict the numbers-in-span pre-filter AND let
  416:     a non-numeric claim spuriously satisfy the numeric->atom map (via
  417:     ``"1" in atom_literal``). The boundary-aware regex is therefore the
  418:     authoritative tokenizer here. The AUTHORITATIVE numeric verdict is still
  419:     delegated to verify_sentence_provenance(require_number_match=True) downstream
  420:     (check 5), which runs strict_verify's full numbers-in-span machinery тАФ this
  421:     pre-filter is a cheap, boundary-correct subset of it.
  422: 
  423:     claim_atom_extractor is imported lazily so this module does not couple to its
  424:     import graph at load. claim_atom_extractor + strict_verify are UNCHANGED.
  425:     """
  426:     from src.polaris_graph.generator.claim_atom_extractor import _NUMBER_ATOM_RE

  429:     for m in _NUMBER_ATOM_RE.finditer(text):
  430:         raw = m.group("value").strip()
  431:         if not raw:
  432:             continue
  433:         nums.add(raw)
  434:         # sign-normalize the unicode minus so "-1.86" and the unicode-minus form
  435:         # both count as the same number.
  436:         nums.add(raw.replace("тИТ", "-"))

  501: def _fuzzy_locate_span(quote: str, source: str) -> Optional[tuple[int, int]]:
  502:     """#1217 Bug B fix (Codex pre-approved candidate (c), CONSTRAINED): recover the
  503:     REAL source window that best matches a PARAPHRASED MAP support_quote, or None.
  504: 
  505:     The MAP model frequently paraphrases (drops markdown italics like ``_S.
  506:     cerevisiae_``; atomizes one source sentence into several) so the quote is not a
  507:     verbatim or whitespace-flexible substring тАФ the live probe on the CDC safety
  508:     source rejected all 3 contraindication findings here ("3 proposed, 0 validated"),
  509:     the exact claims the legacy arm verified. This finds the source window with the
  510:     highest content-word overlap with the quote and ACCEPTS it only if the overlap
  511:     meets a threshold, then SHRINKS to the tight span between the first and last
  512:     matched content word. Faithfulness-safe (Codex constraint): it returns a GENUINE
  513:     source slice тАФ NEVER the model's paraphrase text тАФ and the claim is still
  514:     entailment-checked against it, then the final REDUCE prose is re-checked by the
  515:     UNCHANGED strict_verify, so a wrong recovery can never publish an unsupported
  516:     claim. It only RAISES recall to parity with the final gate's own span binder."""
  517:     from src.polaris_graph.generator.provenance_generator import _content_words
  518: 
  519:     qwords = _content_words(quote)
  520:     if len(qwords) < 2:
  521:         return None  # too little signal to recover safely
  522:     need = max(2, int(_distill_fuzzy_min_overlap_frac() * len(qwords) + 0.9999))
  523: 
  524:     n = len(source)
  525:     window = min(n, max(200, len(quote) * 2))
  526:     stride = max(20, window // 8)
  527: 
  528:     best: Optional[tuple[int, int]] = None
  529:     best_ov = 0
  530:     starts = list(range(0, max(1, n - window + 1), stride))
  531:     starts.append(max(0, n - window))  # always include the tail window
  532:     for i in starts:
  533:         end = min(i + window, n)
  534:         ov = len(qwords & _content_words(source[i:end]))
  535:         if ov > best_ov:
  536:             best_ov = ov
  537:             best = (i, end)
  538:     if best is None or best_ov < need:
  539:         return None
  540: 
  541:     # Localize to the matched content-word region, then EXPAND to the enclosing
  542:     # clause/sentence boundary тАФ do NOT shrink to the first/last content word.
  543:     # #1217 Codex diff-gate P2 (clinical faithfulness): a tight content-word shrink
  544:     # drops leading FUNCTION words, so a source clause "... are not recommended for
  545:     # immunocompromised patients" would shrink to "recommended for immunocompromised
  546:     # patients" тАФ flipping the negation BEFORE the per-finding entailment check sees
  547:     # it. Widening each side to the nearest sentence terminator keeps "not"/"no"/
  548:     # "without"/qualifier words inside the span so the entailment gate judges the TRUE
  549:     # meaning. (The final strict_verify still rebinds against the original evidence
  550:     # row, but the fuzzy gate must not be blind to negation.)
  551:     wstart, wend = best
  552:     win = source[wstart:wend]
  553:     first_off: Optional[int] = None
  554:     last_off: Optional[int] = None
  555:     for m in re.finditer(r"[A-Za-z0-9]+", win):
  556:         if m.group(0).lower() in qwords:
  557:             if first_off is None:
  558:                 first_off = m.start()
  559:             last_off = m.end()
  560:     if first_off is None or last_off is None:
  561:         return best
  562:     # Left boundary: just after the previous sentence terminator (. ! ? or newline).
  563:     left = max(win.rfind(".", 0, first_off), win.rfind("\n", 0, first_off),
  564:                win.rfind("!", 0, first_off), win.rfind("?", 0, first_off))
  565:     start_off = left + 1 if left >= 0 else 0
  566:     while start_off < first_off and win[start_off] in " \t\r\n":
  567:         start_off += 1
  568:     # Right boundary: through the next sentence terminator at/after the last match.
  569:     m_end = re.search(r"[.\n!?]", win[last_off:])
  570:     end_off = last_off + m_end.end() if m_end else len(win)
  571:     return wstart + start_off, wstart + end_off

  574: def _locate_span_with_method(quote: str, source: str) -> Optional[tuple[int, int, str]]:
  575:     """Locate ``quote`` inside ``source``, returning (start, end, method) of the
  576:     REAL source slice, or None. ``method`` тИИ {"exact", "stripped", "whitespace",
  577:     "fuzzy"} тАФ the caller uses it to apply an EXTRA entailment gate to the riskier
  578:     "fuzzy" (paraphrase-recovered) case (#1217).
  579: 
  580:     I-perm-016 / #1217: the MAP model rarely copies a span VERBATIM тАФ it reformats
  581:     whitespace/newlines AND paraphrases (drops markdown italics, atomizes a source
  582:     sentence). An exact-substring-only check rejected ~most findings, collapsing the
  583:     distillate to empty. We recover the real span at four escalating tolerances; the
  584:     output gates (entailment + the final strict_verify) remain the authority on the
  585:     prose, and "fuzzy" recoveries get an additional blocking entailment check in
  586:     `_validate_finding` so a meaning-changing paraphrase (e.g. "all" -> "some",
  587:     negation flips) cannot enter the ledger on content-word overlap alone."""
  588:     if not quote or not source:
  589:         return None
  590:     i = source.find(quote)
  591:     if i >= 0:
  592:         return i, i + len(quote), "exact"
  593:     stripped = quote.strip()
  594:     if stripped and stripped != quote:
  595:         i = source.find(stripped)
  596:         if i >= 0:
  597:             return i, i + len(stripped), "stripped"
  598:     toks = stripped.split()
  599:     if not toks:
  600:         return None
  601:     # Same tokens, any whitespace run between them (newline/space reformatting).
  602:     pattern = r"\s+".join(re.escape(t) for t in toks)
  603:     try:
  604:         m = re.search(pattern, source)
  605:     except re.error:
  606:         m = None
  607:     if m:
  608:         return m.start(), m.end(), "whitespace"
  609:     # #1217 Bug B: exact + whitespace-flexible failed -> the MAP paraphrased the
  610:     # quote (markdown italics dropped, source sentence atomized). Recover the REAL
  611:     # source window by content-word overlap (CONSTRAINED, threshold-gated). Returns a
  612:     # genuine source slice or None; the caller adds a blocking entailment gate.
  613:     fz = _fuzzy_locate_span(stripped, source)
  614:     if fz is not None:
  615:         return fz[0], fz[1], "fuzzy"
  616:     return None

  630: def _validate_finding(
  631:     raw: dict,
  632:     *,
  633:     evidence_id: str,
  634:     direct_quote: str,
  635:     tier: str,
  636:     evidence_pool: dict[str, dict[str, Any]],
  637:     section_atoms: dict[str, Any],
  638:     finding_id: str,
  639:     raw_index: int = -1,
  640: ) -> Optional[DistilledFinding]:
  641:     """Validate one raw MAP finding fail-closed. Returns the DistilledFinding on
  642:     success, or None if it fails a BLOCKING check (the source then falls to a
  643:     validation_failed CoverageRow if no finding survives).
  644: 
  645:     Checks (in order; only steps 1-3 are BLOCKING after the #1217 recall fixes):
  646:       1. (BLOCKING) support_quote is LOCATABLE in direct_quote, tolerant of
  647:          whitespace/newline reformatting (`_locate_span_in_source`); a reworded
  648:          quote is unlocatable -> reject.
  649:       2. Offsets: if supplied offsets already point at a matching occurrence keep
  650:          them; otherwise use the located (recovered) span.
  651:       3. (BLOCKING) direct_quote[start:end] is non-empty.
  652:       4. (NON-BLOCKING, #1217) numbers-in-span is computed for telemetry only; it
  653:          never rejects тАФ strict_verify re-checks numbers against the final span.
  654:       5. Atom mapping: a numeric finding with NO section-local atom is KEPT
  655:          (atom_ids stays []) per the #1209 Codex diff-gate ruling.
  656:       6. Per-finding entailment is computed via the UNCHANGED production verifier.
  657:          It is NON-BLOCKING for exact/whitespace matches; BLOCKING for a FUZZY
  658:          (paraphrase-recovered) span (#1217) тАФ a fuzzy span was matched on
  659:          content-word overlap, blind to meaning-changing function words, so it must
  660:          additionally ENTAIL the claim. A verifier EXCEPTION rejects (fail-closed).
  661:          The FINAL per-sentence strict_verify on the REDUCE prose remains the SOLE
  662:          publication authority for all paths.

  669: 
  670:     def _rej(step: str, reason: str) -> None:
  671:         """#1217 diagnostic: log which validation STEP killed this proposal (keyed on
  672:         evidence_id + raw_index) then return None. Logging only тАФ no behavior change."""
  673:         if _dbg:
  674:             logger.warning(
  675:                 "[DISTILL_DEBUG] REJECT ev=%s raw_index=%d step=%s reason=%s claim=%r",
  676:                 evidence_id, raw_index, step, reason,
  677:                 str(raw.get("claim", ""))[:90],
  678:             )
  679:         return None
  680: 
  681:     claim = str(raw.get("claim", "") or "").strip()
  682:     support_quote = str(raw.get("support_quote", "") or "")
  683:     if not claim or not support_quote:
  684:         return _rej("step0_empty", "empty claim or support_quote")
  685: 
  686:     # (1) Locate support_quote in direct_quote, TOLERANT of whitespace/newline
  687:     # reformatting (the MAP model rarely copies verbatim). On a hit we ADOPT the
  688:     # real source slice as support_quote below тАФ faithfulness-safe (same words,
  689:     # real offsets). A reworded quote stays unlocatable -> reject. (#1217: the
  690:     # prior exact-substring-only check rejected ~most findings -> empty distillate
  691:     # -> section collapse. The MAP filter is an EXTRACTION filter; strict_verify +
  692:     # the 4-role gate remain the authority on the final prose.)
  693:     located = _locate_span_with_method(support_quote, direct_quote)
  694:     if located is None:
  695:         return _rej("step1_locate", "support_quote not locatable in direct_quote (reworded/paraphrased)")
  696:     loc_start, loc_end, locate_method = located
  697: 
  698:     # (2) Offset reconciliation: prefer the model's supplied offsets ONLY if they
  699:     # already point at the EXACT support_quote (handles duplicate occurrences);
  700:     # otherwise use the located (recovered) span.
  701:     try:
  702:         supplied_start = int(raw.get("span_start", -1))
  703:         supplied_end = int(raw.get("span_end", -1))
  704:     except (TypeError, ValueError):

  709:         and direct_quote[supplied_start:supplied_end] == support_quote
  710:     ):

  712:         locate_method = "exact"  # model offsets pointed at the exact quote
  713:     else:

  715: 
  716:     # (3) Adopt the REAL source slice as support_quote (this is what the
  717:     # whitespace-flexible recovery returns) and re-assert it is non-empty.
  718:     support_quote = direct_quote[span_start:span_end]
  719:     if not support_quote.strip():
  720:         return _rej("step3_empty_slice", "recovered source slice is empty/whitespace")
  721: 
  722:     # Parse declared numbers/entities defensively.
  723:     raw_numbers = raw.get("numbers") or []
  724:     numbers = [str(n) for n in raw_numbers if str(n).strip()]
  725:     raw_entities = raw.get("entities") or []
  726:     entities = [str(e) for e in raw_entities if str(e).strip()]

  406: def _numbers_in_text(text: str) -> set[str]:
  407:     """Numeric tokens under the boundary-aware definition shared with the
  408:     downstream verifier (spec section 1 check 4: "the same numeric definitions
  409:     as strict_verify plus claim_atom_extractor").
  410: 
  411:     Uses claim_atom_extractor._NUMBER_ATOM_RE, whose ``(?<![A-Za-z0-9_.])``
  412:     lookbehind correctly excludes the ``1`` embedded in "HbA1c" (preceded by a
  413:     letter) while still catching ``-1.86``, ``99.9``, and ranges. The bare
  414:     strict_verify._decimals pattern (no boundary) over-tokenizes "HbA1c" into a
  415:     junk ``1`` that would both over-strict the numbers-in-span pre-filter AND let
  416:     a non-numeric claim spuriously satisfy the numeric->atom map (via
  417:     ``"1" in atom_literal``). The boundary-aware regex is therefore the
  418:     authoritative tokenizer here. The AUTHORITATIVE numeric verdict is still
  419:     delegated to verify_sentence_provenance(require_number_match=True) downstream
  420:     (check 5), which runs strict_verify's full numbers-in-span machinery тАФ this
  421:     pre-filter is a cheap, boundary-correct subset of it.
  422: 
  423:     claim_atom_extractor is imported lazily so this module does not couple to its
  424:     import graph at load. claim_atom_extractor + strict_verify are UNCHANGED.
  425:     """
  426:     from src.polaris_graph.generator.claim_atom_extractor import _NUMBER_ATOM_RE
  427: 
  428:     nums: set[str] = set()
  429:     for m in _NUMBER_ATOM_RE.finditer(text):
  430:         raw = m.group("value").strip()
  431:         if not raw:
  432:             continue
  433:         nums.add(raw)
  434:         # sign-normalize the unicode minus so "-1.86" and the unicode-minus form
  435:         # both count as the same number.
  436:         nums.add(raw.replace("тИТ", "-"))
  437:     return nums

  444:     appear in the support span. One-way constraint, identical in spirit to
  445:     strict_verify's numeric_mismatch check. The authoritative numeric verdict is
  446:     still delegated to verify_sentence_provenance(require_number_match=True); this

  501: def _fuzzy_locate_span(quote: str, source: str) -> Optional[tuple[int, int]]:
  502:     """#1217 Bug B fix (Codex pre-approved candidate (c), CONSTRAINED): recover the
  503:     REAL source window that best matches a PARAPHRASED MAP support_quote, or None.
  504: 
  505:     The MAP model frequently paraphrases (drops markdown italics like ``_S.
  506:     cerevisiae_``; atomizes one source sentence into several) so the quote is not a
  507:     verbatim or whitespace-flexible substring тАФ the live probe on the CDC safety
  508:     source rejected all 3 contraindication findings here ("3 proposed, 0 validated"),
  509:     the exact claims the legacy arm verified. This finds the source window with the
  510:     highest content-word overlap with the quote and ACCEPTS it only if the overlap
  511:     meets a threshold, then SHRINKS to the tight span between the first and last
  512:     matched content word. Faithfulness-safe (Codex constraint): it returns a GENUINE
  513:     source slice тАФ NEVER the model's paraphrase text тАФ and the claim is still
  514:     entailment-checked against it, then the final REDUCE prose is re-checked by the
  515:     UNCHANGED strict_verify, so a wrong recovery can never publish an unsupported
  516:     claim. It only RAISES recall to parity with the final gate's own span binder."""
  517:     from src.polaris_graph.generator.provenance_generator import _content_words
  518: 
  519:     qwords = _content_words(quote)
  520:     if len(qwords) < 2:
  521:         return None  # too little signal to recover safely
  522:     need = max(2, int(_distill_fuzzy_min_overlap_frac() * len(qwords) + 0.9999))
  523: 
  524:     n = len(source)
  525:     window = min(n, max(200, len(quote) * 2))
  526:     stride = max(20, window // 8)
  527: 
  528:     best: Optional[tuple[int, int]] = None
  529:     best_ov = 0
  530:     starts = list(range(0, max(1, n - window + 1), stride))
  531:     starts.append(max(0, n - window))  # always include the tail window
  532:     for i in starts:
  533:         end = min(i + window, n)
  534:         ov = len(qwords & _content_words(source[i:end]))
  535:         if ov > best_ov:
  536:             best_ov = ov
  537:             best = (i, end)
  538:     if best is None or best_ov < need:
  539:         return None
  540: 
  541:     # Localize to the matched content-word region, then EXPAND to the enclosing
  542:     # clause/sentence boundary тАФ do NOT shrink to the first/last content word.
  543:     # #1217 Codex diff-gate P2 (clinical faithfulness): a tight content-word shrink
  544:     # drops leading FUNCTION words, so a source clause "... are not recommended for
  545:     # immunocompromised patients" would shrink to "recommended for immunocompromised
  546:     # patients" тАФ flipping the negation BEFORE the per-finding entailment check sees
  547:     # it. Widening each side to the nearest sentence terminator keeps "not"/"no"/
  548:     # "without"/qualifier words inside the span so the entailment gate judges the TRUE
  549:     # meaning. (The final strict_verify still rebinds against the original evidence
  550:     # row, but the fuzzy gate must not be blind to negation.)
  551:     wstart, wend = best
  552:     win = source[wstart:wend]
  553:     first_off: Optional[int] = None
  554:     last_off: Optional[int] = None
  555:     for m in re.finditer(r"[A-Za-z0-9]+", win):
  556:         if m.group(0).lower() in qwords:
  557:             if first_off is None:
  558:                 first_off = m.start()
  559:             last_off = m.end()
  560:     if first_off is None or last_off is None:
  561:         return best
  562:     # Left boundary: just after the previous sentence terminator (. ! ? or newline).
  563:     left = max(win.rfind(".", 0, first_off), win.rfind("\n", 0, first_off),
  564:                win.rfind("!", 0, first_off), win.rfind("?", 0, first_off))
  565:     start_off = left + 1 if left >= 0 else 0
  566:     while start_off < first_off and win[start_off] in " \t\r\n":
  567:         start_off += 1
  568:     # Right boundary: through the next sentence terminator at/after the last match.
  569:     m_end = re.search(r"[.\n!?]", win[last_off:])
  570:     end_off = last_off + m_end.end() if m_end else len(win)
  571:     return wstart + start_off, wstart + end_off

  574: def _locate_span_with_method(quote: str, source: str) -> Optional[tuple[int, int, str]]:
  575:     """Locate ``quote`` inside ``source``, returning (start, end, method) of the
  576:     REAL source slice, or None. ``method`` тИИ {"exact", "stripped", "whitespace",
  577:     "fuzzy"} тАФ the caller uses it to apply an EXTRA entailment gate to the riskier
  578:     "fuzzy" (paraphrase-recovered) case (#1217).
  579: 
  580:     I-perm-016 / #1217: the MAP model rarely copies a span VERBATIM тАФ it reformats
  581:     whitespace/newlines AND paraphrases (drops markdown italics, atomizes a source
  582:     sentence). An exact-substring-only check rejected ~most findings, collapsing the
  583:     distillate to empty. We recover the real span at four escalating tolerances; the
  584:     output gates (entailment + the final strict_verify) remain the authority on the
  585:     prose, and "fuzzy" recoveries get an additional blocking entailment check in
  586:     `_validate_finding` so a meaning-changing paraphrase (e.g. "all" -> "some",
  587:     negation flips) cannot enter the ledger on content-word overlap alone."""
  588:     if not quote or not source:
  589:         return None
  590:     i = source.find(quote)
  591:     if i >= 0:
  592:         return i, i + len(quote), "exact"
  593:     stripped = quote.strip()
  594:     if stripped and stripped != quote:
  595:         i = source.find(stripped)
  596:         if i >= 0:
  597:             return i, i + len(stripped), "stripped"
  598:     toks = stripped.split()
  599:     if not toks:
  600:         return None
  601:     # Same tokens, any whitespace run between them (newline/space reformatting).
  602:     pattern = r"\s+".join(re.escape(t) for t in toks)
  603:     try:
  604:         m = re.search(pattern, source)
  605:     except re.error:
  606:         m = None
  607:     if m:
  608:         return m.start(), m.end(), "whitespace"
  609:     # #1217 Bug B: exact + whitespace-flexible failed -> the MAP paraphrased the
  610:     # quote (markdown italics dropped, source sentence atomized). Recover the REAL
  611:     # source window by content-word overlap (CONSTRAINED, threshold-gated). Returns a
  612:     # genuine source slice or None; the caller adds a blocking entailment gate.
  613:     fz = _fuzzy_locate_span(stripped, source)
  614:     if fz is not None:
  615:         return fz[0], fz[1], "fuzzy"
  616:     return None

  637:     section_atoms: dict[str, Any],

  639:     raw_index: int = -1,

  641:     """Validate one raw MAP finding fail-closed. Returns the DistilledFinding on
  642:     success, or None if it fails a BLOCKING check (the source then falls to a
  643:     validation_failed CoverageRow if no finding survives).
  644: 
  645:     Checks (in order; only steps 1-3 are BLOCKING after the #1217 recall fixes):
  646:       1. (BLOCKING) support_quote is LOCATABLE in direct_quote, tolerant of
  647:          whitespace/newline reformatting (`_locate_span_in_source`); a reworded

  649:       2. Offsets: if supplied offsets already point at a matching occurrence keep
  650:          them; otherwise use the located (recovered) span.
  651:       3. (BLOCKING) direct_quote[start:end] is non-empty.
  652:       4. (NON-BLOCKING, #1217) numbers-in-span is computed for telemetry only; it
  653:          never rejects тАФ strict_verify re-checks numbers against the final span.
  654:       5. Atom mapping: a numeric finding with NO section-local atom is KEPT
  655:          (atom_ids stays []) per the #1209 Codex diff-gate ruling.
  656:       6. Per-finding entailment is computed via the UNCHANGED production verifier.
  657:          It is NON-BLOCKING for exact/whitespace matches; BLOCKING for a FUZZY
  658:          (paraphrase-recovered) span (#1217) тАФ a fuzzy span was matched on
  659:          content-word overlap, blind to meaning-changing function words, so it must
  660:          additionally ENTAIL the claim. A verifier EXCEPTION rejects (fail-closed).
  661:          The FINAL per-sentence strict_verify on the REDUCE prose remains the SOLE
  662:          publication authority for all paths.
  663: 
  664:     raw_index: the 0-based index of this finding in the MAP `findings` array, used
  665:     for #1217 PG_DISTILL_DEBUG reject tracing (rejected findings do NOT consume a
  666:     finding_id, so finding_id alone cannot disambiguate which proposal was killed).
  667:     """
  668:     _dbg = os.getenv("PG_DISTILL_DEBUG", "").strip().lower() in ("1", "true", "yes", "on")
  669: 
  670:     def _rej(step: str, reason: str) -> None:
  671:         """#1217 diagnostic: log which validation STEP killed this proposal (keyed on
  672:         evidence_id + raw_index) then return None. Logging only тАФ no behavior change."""
  673:         if _dbg:
  674:             logger.warning(
  675:                 "[DISTILL_DEBUG] REJECT ev=%s raw_index=%d step=%s reason=%s claim=%r",
  676:                 evidence_id, raw_index, step, reason,
  677:                 str(raw.get("claim", ""))[:90],
  678:             )

  681:     claim = str(raw.get("claim", "") or "").strip()
  682:     support_quote = str(raw.get("support_quote", "") or "")

  686:     # (1) Locate support_quote in direct_quote, TOLERANT of whitespace/newline
  687:     # reformatting (the MAP model rarely copies verbatim). On a hit we ADOPT the
  688:     # real source slice as support_quote below тАФ faithfulness-safe (same words,
  689:     # real offsets). A reworded quote stays unlocatable -> reject. (#1217: the
  690:     # prior exact-substring-only check rejected ~most findings -> empty distillate
  691:     # -> section collapse. The MAP filter is an EXTRACTION filter; strict_verify +
  692:     # the 4-role gate remain the authority on the final prose.)
  693:     located = _locate_span_with_method(support_quote, direct_quote)
  694:     if located is None:
  695:         return _rej("step1_locate", "support_quote not locatable in direct_quote (reworded/paraphrased)")
  696:     loc_start, loc_end, locate_method = located
  697: 
  698:     # (2) Offset reconciliation: prefer the model's supplied offsets ONLY if they
  699:     # already point at the EXACT support_quote (handles duplicate occurrences);
  700:     # otherwise use the located (recovered) span.

 succeeded in 9059ms:
  701:     try:
  702:         supplied_start = int(raw.get("span_start", -1))
  703:         supplied_end = int(raw.get("span_end", -1))
  704:     except (TypeError, ValueError):
  705:         supplied_start, supplied_end = -1, -1

  707:     if (
  708:         0 <= supplied_start < supplied_end <= len(direct_quote)

  712:         locate_method = "exact"  # model offsets pointed at the exact quote
  713:     else:
  714:         span_start, span_end = loc_start, loc_end

  716:     # (3) Adopt the REAL source slice as support_quote (this is what the
  717:     # whitespace-flexible recovery returns) and re-assert it is non-empty.
  718:     support_quote = direct_quote[span_start:span_end]
  719:     if not support_quote.strip():
  720:         return _rej("step3_empty_slice", "recovered source slice is empty/whitespace")
  721: 
  722:     # Parse declared numbers/entities defensively.
  723:     raw_numbers = raw.get("numbers") or []
  724:     numbers = [str(n) for n in raw_numbers if str(n).strip()]
  725:     raw_entities = raw.get("entities") or []
  726:     entities = [str(e) for e in raw_entities if str(e).strip()]

  728:     contradiction_key = str(raw.get("contradiction_key", "") or "").strip()

  732:     # Claude AND Codex independent re-forensics AGREE on candidate (b)). This was a
  733:     # hard `return None` reject; it collapsed distill recall below legacy on the

  738:     # (`_find_best_span_for_sentence`, live_deepseek_generator.py:244) and re-runs
  739:     # strict_verify `require_number_match=True` on it. Concrete killer (Codex): the
  740:     # CDC stat "14 (95% CI 4-44)" tokenizes "4-44" as ONE range token here, while the

  742:     # finding is rejected before REDUCE ever sees it тАФ even though final strict_verify
  743:     # normalizes the range dash and would accept/drop the published span correctly.
  744:     # Its own docstring states this is only a "cheap local pre-filter ... before the
  745:     # (more expensive) entailment call"; entailment is already non-blocking at step
  746:     # (6) below, so the pre-filter's sole rationale is dead. Mirrors the step-(6)
  747:     # treatment exactly: compute the boolean for telemetry, NEVER gate on it.
  748:     # Faithfulness UNCHANGED: strict_verify on the final REDUCE prose stays the SOLE
  749:     # publication authority (numbers-in-span AND >=2 content-word overlap AND
  750:     # enforce-mode entailment); 4-role / D8 byte-untouched.
  751:     numbers_in_span = _all_numbers_in_span(claim, numbers, support_quote)
  752:     if not numbers_in_span:

  754:             "[evidence_distiller] finding %s/%s: declared numbers not all inside the "
  755:             "narrow support_quote (KEPT тАФ non-blocking per #1217; the final "
  756:             "strict_verify re-checks numbers against the prose-matched span)",

  758:         )

  762:     atom_ids: list[str] = []

  764:         atom_ids = _match_atom_ids(
  765:             evidence_id=evidence_id,
  766:             numbers=claim_numbers,
  767:             section_atoms=section_atoms,
  768:         )
  769:         # I-perm-016 (#1209) Codex diff-gate iter-1 P1 ruling: KEEP a numeric
  770:         # finding that has NO matching section-local atom (atom_ids stays []),
  771:         # do NOT reject. `claim_atom_extractor` routes each minted atom to a
  772:         # SINGLE primary_section, so a span-exact + numbers-in-span + ENTAILED
  773:         # finding in a non-matching section would otherwise be shed solely on a
  774:         # cataloguing artifact ├втВмтАЭ lowering verified coverage on the safety/
  775:         # incidence-heavy drb_76 replay (the OPPOSITE of this keystone's goal).
  776:         # Faithfulness is unaffected: step (6) below still gates the finding
  777:         # through the UNCHANGED production verifier (numbers-in-span AND
  778:         # span-entails-claim), and the final REDUCE prose is re-checked by the
  779:         # unchanged strict_verify. A kept empty-atom_ids finding renders NO
  780:         # (atom_NNN) marker, so the default-OFF atom_refusal_validator treats it
  781:         # as narrative-only. (Codex: "choose (b) ... faithfulness-safe".)
  782: 
  783:     # (6) Per-finding entailment via the UNCHANGED production verifier path тАФ
  784:     # BLOCKING for FUZZY recoveries ONLY, and the verifier is CALLED only for them.
  785:     # A fuzzy span was matched on content-word OVERLAP, blind to meaning-changing
  786:     # function words ("all"->"some", negation flips), so it must additionally ENTAIL
  787:     # the claim. Exact/whitespace matches are verbatim source text that cannot drift
  788:     # in meaning, so their per-finding entailment was NON-BLOCKING тАФ and #1217 Codex
  789:     # diff-gate P2 (perf) flagged that calling verify_sentence_provenance for EVERY
  790:     # finding made MAXEV=40 prohibitively slow. We therefore SKIP the (slow) verifier
  791:     # call entirely for exact/whitespace, since its verdict never gated them. The
  792:     # final per-sentence strict_verify on the REDUCE prose remains the SOLE
  793:     # publication authority for ALL paths.
  794:     finding_entailed = True  # exact/whitespace: verbatim, gated downstream by strict_verify
  795:     if locate_method == "fuzzy":
  796:         probe = f"{claim} [#ev:{evidence_id}:{span_start}-{span_end}]"
  797:         try:
  798:             verdict = verify_sentence_provenance(
  799:                 probe, evidence_pool, require_number_match=True,
  800:             )
  801:         except Exception as exc:  # noqa: BLE001 тАФ fail closed on any verifier error
  802:             logger.warning(
  803:                 "[evidence_distiller] verifier raised on finding %s/%s: %s тАФ rejecting",
  804:                 evidence_id, finding_id, exc,
  805:             )
  806:             return _rej("step6_verifier_error", f"verify_sentence_provenance raised: {exc}")
  807:         finding_entailed = bool(getattr(verdict, "is_verified", False))
  808:         if not finding_entailed:
  809:             return _rej("step6_fuzzy_not_entailed",
  810:                         "fuzzy-recovered (paraphrased) span does not entail the claim")
  811:     if _dbg:
  812:         logger.warning(
  813:             "[DISTILL_DEBUG] KEPT ev=%s raw_index=%d finding_id=%s method=%s entailed=%s numbers_in_span=%s claim=%r",
  814:             evidence_id, raw_index, finding_id, locate_method, finding_entailed, numbers_in_span,
  815:             claim[:90],
  816:         )
  817: 
  818:     return DistilledFinding(
  819:         finding_id=finding_id,
  820:         evidence_id=evidence_id,
  821:         claim=claim,
  822:         span_start=span_start,
  823:         span_end=span_end,
  824:         support_quote=support_quote,
  825:         numbers=numbers,
  826:         entities=entities,
  827:         caveat=caveat,
  828:         contradiction_key=contradiction_key,
  829:         source_tier=source_tier,
  830:         atom_ids=atom_ids,
  831:     )

  838: def _atom_rows_for_evidence(
  839:     evidence_id: str, section_atoms: dict[str, Any],
  840: ) -> str:
  841:     """Render the section-local numeric atoms for ONE evidence_id as compact rows
  842:     for the MAP prompt's NUMERIC_ATOMS_AVAILABLE block."""
  843:     rows: list[str] = []
  844:     for aid, atom in sorted(section_atoms.items()):
  845:         if getattr(atom, "evidence_id", "") != evidence_id:
  846:             continue
  847:         value = getattr(atom, "value", "")
  848:         unit = getattr(atom, "unit", "")
  849:         endpoint = getattr(atom, "endpoint", "")
  850:         entity = getattr(atom, "entity", "")
  851:         rows.append(
  852:             f"{aid}: value={value}{(' ' + unit) if unit else ''} "
  853:             f"endpoint={endpoint}"
  854:             + (f" entity={entity}" if entity else "")
  855:         )
  856:     return "\n".join(rows) if rows else "(none)"
