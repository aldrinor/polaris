## §C Red-team checklist

1. **pako vs tar-stream** — pako handles gzip (mature, browser-tested, GitHub uses it); tar-stream is the standard Node tar parser. `tar-stream@3.x` is browser-build-compatible via Webpack/Next bundling.
2. **SHA-256 verification** — uses `crypto.subtle.digest("SHA-256", ...)` (browser-native since 2016). Result is base16 hex matching the manifest hash format.
3. **ArrayBuffer typing** — `bytes.buffer.slice(...)` ensures we pass `ArrayBuffer` (not `SharedArrayBuffer`) per TypeScript's strict BufferSource check.
4. **`_findFile` suffix-match** — handles both `tar -czf bundle.tar.gz <dir>` (entries prefixed with `<dir>/`) and `tar -czf bundle.tar.gz -C <dir> .` (entries without prefix).
5. **Filename traversal safety** — `_findFile` does NOT use the entry name directly for fs access (we're in-browser); manifest declares paths; no traversal vector.
6. **`runId` derivation** — `BundleMetadata` does not carry run_id in the v1.0 frozen schema; falls back to filename stem. Documented in loader.
7. **Tab IDs** — added "metadata" tab. Existing Tabs component uses `value=` keys; no collision.
8. **MetadataPanel data-testid** — 5 specific testids let Playwright assert each field individually.
9. **`InspectorView` import from offline/** — page.tsx imports `@/app/inspector/[runId]/inspector_view`. Next.js App Router permits this; component is "use client" already.
10. **Error code surfacing** — `BundleClientLoaderError.code` ("ungzip_failed", "manifest_missing", "sha256_mismatch", etc.) is included in the error card for debugging.

 127:     sanitize aggressively: only [a-zA-Z0-9_-]. Anything else becomes _.
 128:     Length-capped at 100 chars to avoid pathological inputs.
 129:     """
 130:     safe = "".join(
 131:         c if c.isalnum() or c in "-_" else "_" for c in source_id
 132:     )[:100]
 133:     return f"{SOURCES_DIR}/{safe}.txt"

 136: def build_manifest_and_files(
 137:     decision: ScopeDecision,
 138:     pool: EvidencePool,
 139:     report: VerifiedReport,
 140:     *,
 141:     extra_files: dict[str, tuple[bytes, str]] | None = None,
 142: ) -> tuple[BundleManifest, dict[str, bytes]]:
 143:     """Build the bundle manifest + content-file dict.
 144: 
 145:     Args:
 146:         extra_files: optional {path: (content_bytes, content_type)} of
 147:             caller-supplied artifacts to include + hash in the signed
 148:             manifest — I-gen-004 (#496) threads the run's
 149:             reasoning_trace.jsonl through here. Each path must NOT
 150:             collide with a core bundle file.
 151: 
 152:     Returns:
 153:         (manifest, files) where files is {path: content_bytes}. The
 154:         caller (bundle_builder) packs files into a tarball and writes
 155:         the manifest.yaml + manifest.yaml.asc alongside.
 156: 
 157:     Raises:
 158:         ValueError if report.pipeline_verdict != 'success', or if an
 159:         extra_files path collides with a core bundle file.
 160:     """
 161:     if report.pipeline_verdict != "success":
 162:         raise ValueError(
 163:             f"cannot build audit bundle for verdict "
 164:             f"{report.pipeline_verdict!r}; report must be successful"
 165:         )
 166: 
 167:     files_bytes: dict[str, bytes] = {}
 168: 
 169:     # 1. Slice 001 ScopeDecision
 170:     sd_bytes = _serialize_json_canonical(decision)
 171:     files_bytes[FILE_SCOPE_DECISION] = sd_bytes
 172: 
 173:     # 2. Slice 002 EvidencePool
 174:     ep_bytes = _serialize_json_canonical(pool)
 175:     files_bytes[FILE_EVIDENCE_POOL] = ep_bytes
 176: 
 177:     # 3. Slice 003 VerifiedReport
 178:     vr_bytes = _serialize_json_canonical(report)
 179:     files_bytes[FILE_VERIFIED_REPORT] = vr_bytes
 180: 
 181:     # 4. Sovereignty: refuse to ship verbatim spans for non-cleared sources
 182:     assert_all_pool_sources_legal_cleared(pool)
 183: 
 184:     # 4b. Per-source snapshots + cited-span reachability guard
 185:     snapshot_entries = snapshot_sources_with_reachable(report, pool)
 186:     _assert_cited_spans_reachable(report, snapshot_entries)
 187:     for source_id, entry in snapshot_entries.items():
 188:         files_bytes[_safe_source_filename(source_id)] = entry.text.encode("utf-8")
 189:     snapshots = {sid: e.text for sid, e in snapshot_entries.items()}
 190: 
 191:     # 4b. REVIEWER_README ships verbatim in every bundle as metadata content_type.
 192:     readme_path = Path(__file__).parent / "REVIEWER_README.md"
 193:     files_bytes[FILE_REVIEWER_README] = readme_path.read_bytes()
 194: 
 195:     # 5. Bundle metadata (versions + creation timestamp)
 196:     metadata = {
 197:         "bundle_version": BUNDLE_VERSION,
 198:         "polaris_version": POLARIS_VERSION,
 199:         "created_at_utc": datetime.now(timezone.utc).isoformat().replace(
 200:             "+00:00", "Z"
 201:         ),
 202:         "generator_model": report.generator_model,
 203:         "decision_id": decision.decision_id,
 204:         "pool_id": report.pool_id,
 205:         "report_id": report.report_id,
 206:         "source_snapshot_count": len(snapshots),
 207:     }
 208:     files_bytes[FILE_METADATA] = json.dumps(
 209:         metadata,
 210:         sort_keys=True,
 211:         ensure_ascii=False,
 212:         separators=(",", ":"),
 213:     ).encode("utf-8")
 214: 
 215:     # I-gen-004 (#496): merge caller-supplied extra files (e.g. the run's
 216:     # reasoning_trace.jsonl). Each carries its own content_type; a path
 217:     # collision with a core bundle file is a hard error.
 218:     extra_content_types: dict[str, str] = {}
 219:     for path, (content, ct_extra) in (extra_files or {}).items():
 220:         if path in files_bytes:
 221:             raise ValueError(
 222:                 f"extra_files path {path!r} collides with a core bundle file"
 223:             )
 224:         if path in _RESERVED_TAR_MEMBERS:
 225:             raise ValueError(
 226:                 f"extra_files path {path!r} collides with a reserved tar "
 227:                 f"member ({sorted(_RESERVED_TAR_MEMBERS)} are written "
 228:                 f"alongside the packed files)"
 229:             )
 230:         files_bytes[path] = content
 231:         extra_content_types[path] = ct_extra
 232: 
 233:     # Build FileEntry list with content_type tags
 234:     file_entries: list[FileEntry] = []
 235:     content_type_by_path = {
 236:         FILE_SCOPE_DECISION: "scope_decision",
 237:         FILE_EVIDENCE_POOL: "evidence_pool",
 238:         FILE_VERIFIED_REPORT: "verified_report",
 239:         FILE_METADATA: "metadata",
 240:     }
 241:     for path, content in files_bytes.items():
 242:         ct = (
 243:             content_type_by_path.get(path)
 244:             or extra_content_types.get(path)
 245:             or (
 246:                 "source_snapshot"
 247:                 if path.startswith(SOURCES_DIR + "/")
 248:                 else "metadata"
 249:             )
 250:         )
 251:         file_entries.append(
 252:             FileEntry(
 253:                 path=path,
 254:                 sha256=_sha256_bytes(content),
 255:                 size_bytes=len(content),
 256:                 content_type=ct,  # type: ignore[arg-type]
 257:             )
 258:         )
 259: 
 260:     # Sort entries deterministically for stable manifest output
 261:     file_entries.sort(key=lambda f: f.path)
 262: 
 263:     manifest = BundleManifest(
 264:         decision_id=decision.decision_id,
 265:         pool_id=report.pool_id,
 266:         report_id=report.report_id,
 267:         generator_model=report.generator_model,
 268:         polaris_version=POLARIS_VERSION,
 269:         files=file_entries,
 270:     )
 271:     return manifest, files_bytes

 274: def serialize_manifest_yaml(manifest: BundleManifest) -> bytes:
 275:     """Produce canonical YAML bytes for GPG signing.
 276: 
 277:     PyYAML default_flow_style=False + sort_keys=True keeps it stable
 278:     across re-serializations.
 279:     """
 280:     payload = manifest.model_dump(mode="json")
 281:     return yaml.safe_dump(
 282:         payload,
 283:         default_flow_style=False,
 284:         sort_keys=True,
 285:         allow_unicode=True,
 286:     ).encode("utf-8")
