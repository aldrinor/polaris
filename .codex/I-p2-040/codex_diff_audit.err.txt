  title = "Couldn't load this",
  message,
  onRetry,
}: {
  title?: string;
  /** REQUIRED — the specific, operational error (never generic). */
  message: string;
  onRetry?: () => void;
}) {

    if (
      process.env.NODE_ENV !== "production" &&

        <button
