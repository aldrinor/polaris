const FRENCH_ACCENTED_RE = /[éèêëàâäçîïôöùûüÿñ]/i;

function looksNonEnglish(s: string): boolean {
  if (FRENCH_ACCENTED_RE.test(s)) return true;
  const matches = s.match(FRENCH_STOPWORD_RE);
  return (matches?.length ?? 0) >= 3;
}
