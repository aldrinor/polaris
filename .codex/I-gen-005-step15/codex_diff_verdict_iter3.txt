   67: def _round_dollar_to_bucket(amount: float, tolerance: float = 0.05) -> int:
   68:     """Round a dollar amount to a ±5% bucket for signature comparison.
   69: 
   70:     Example: $11.2B and $11.4B both round to the same bucket (~$11B).
   71:               $11.2B and $13.4B fall in different buckets (gap > 5%).
   72:     """
   73:     if amount <= 0:
   74:         return 0
   75:     # log-space bucketing: round to nearest 5% step
   76:     import math
   77:     log_amount = math.log10(amount)
   78:     step = math.log10(1 + tolerance)
   79:     bucket_idx = round(log_amount / step)
   80:     # Return the bucket as an int (representative midpoint)
   81:     return int(round(10 ** (bucket_idx * step)))

  101: def extract_signature(sentence: str) -> FactSignature:
  102:     """Extract a FactSignature from a sentence.
  103: 
  104:     Decimals: floats from percentages.
  105:     Dollar buckets: dollar amounts bucketed by ±5% (after unit normalization).
  106:     Years: 4-digit 19xx/20xx integers.
  107:     """
  108:     decimals: set[float] = set()
  109:     for match in _PERCENT_RE.finditer(sentence):
  110:         try:
  111:             decimals.add(round(float(match.group(1)), 2))
  112:         except (ValueError, TypeError):
  113:             continue
  114: 
  115:     dollar_buckets: set[int] = set()
  116:     for match in _DOLLAR_RE.finditer(sentence):
  117:         try:
  118:             amount = float(match.group(1).replace(",", ""))
  119:         except (ValueError, TypeError):
  120:             continue
  121:         unit = (match.group(2) or "").lower()
  122:         scale = _DOLLAR_UNIT_SCALE.get(unit, 1)
  123:         normalized = amount * scale
  124:         dollar_buckets.add(_round_dollar_to_bucket(normalized))

  129:             years.add(int(match.group(1)))

  134:         decimals=frozenset(decimals),
  135:         dollar_buckets=frozenset(dollar_buckets),
  136:         years=frozenset(years),

  162: def _signatures_overlap(

  192:     # FactSignature(dollar_buckets={204}, decimals={}, years={}) for

  227:     shared_dollars = a.dollar_buckets & b.dollar_buckets

  233: def _signatures_conflict(a: FactSignature, b: FactSignature) -> bool:
  234:     """True iff a populated context axis (years OR dollar_buckets) is
  235:     populated on BOTH sides AND has zero intersection.
  236: 
  237:     Codex brief-iter-1 P1 fix: this is the same populated-axis-conflict
  238:     check used by `_signatures_overlap` top-level guard, lifted into a
  239:     helper so cluster-membership logic can reject contextless-bridge
  240:     merges (e.g. A={2014} and C={2016} merged via B={} bridge).
  241:     """
  242:     if a.is_empty() or b.is_empty():
  243:         return False
  244:     years_conflict = (a.years and b.years) and not (a.years & b.years)
  245:     dollars_conflict = (
  246:         (a.dollar_buckets and b.dollar_buckets)
  247:         and not (a.dollar_buckets & b.dollar_buckets)
  248:     )
  249:     return years_conflict or dollars_conflict

  252: def _cluster_is_compatible(
  253:     candidate: SentenceLocation, cluster: list[SentenceLocation],
  254: ) -> bool:
  255:     """True iff candidate overlaps with AT LEAST ONE cluster member AND
  256:     does NOT conflict with ANY cluster member.
  257: 
  258:     Phase 3 P1-2 fix (Codex review): previous greedy implementation
  259:     compared candidate only against cluster[0]. Comparing against ALL
  260:     members captures transitive chains (A↔B↔C where A↛C directly).
  261: 
  262:     Codex brief-iter-1 P1 fix: requiring "no conflict with any member"
  263:     blocks the contextless-bridge merge — e.g. A={9.2%,13.9%,2014} and
  264:     C={9.2%,13.9%,2016} would otherwise be merged via B={9.2%,13.9%}
  265:     (which overlaps with both and conflicts with neither).
  266:     """
  267:     has_overlap = False
  268:     for member in cluster:
  269:         if _signatures_conflict(candidate.signature, member.signature):
  270:             return False
  271:         if _signatures_overlap(candidate.signature, member.signature):
  272:             has_overlap = True
  273:     return has_overlap

  276: def _clusters_pairwise_compatible(
  277:     cluster_a: list[SentenceLocation], cluster_b: list[SentenceLocation],
  278: ) -> bool:
  279:     """True iff no member of cluster_a conflicts with any member of
  280:     cluster_b.
  281: 
  282:     Codex brief-iter-1 P1 fix: when a candidate bridges two existing
  283:     clusters X and Y, the candidate may be individually compatible with
  284:     both (`_cluster_is_compatible` True for each) while X and Y
  285:     themselves contain mutually-conflicting members. Without this
  286:     pairwise check, the candidate would silently fuse incompatible
  287:     clusters through its contextless overlap.
  288:     """
  289:     for a in cluster_a:
  290:         for b in cluster_b:
  291:             if _signatures_conflict(a.signature, b.signature):
  292:                 return False
  293:     return True

  296: def build_groups(
  297:     sections: dict[str, list[str]],
  298:     section_order: Optional[list[str]] = None,
  299: ) -> list[RedundancyGroup]:
  300:     """Group sentences across sections by overlap-matched FactSignature.
  301: 
  302:     Per Codex Phase 3 diagnosis (`.codex/I-gen-002-phase3/`): the
  303:     earlier exact-FactSignature grouping was too strict — incidental
  304:     years/dollars added in one section but not another broke the match.
  305:     Now uses `_signatures_overlap` core-fact matcher to group
  306:     semantically equivalent sentences even when contextual numerics
  307:     differ.

  309:     Args:
  310:         sections: dict mapping section_title -> list of sentence strings.
  311:         section_order: optional explicit ordering for PRIMARY selection.
  312:             If None, falls back to insertion order of `sections` keys.
  313:

  315:         List of RedundancyGroup, one per overlap-cluster appearing in
  316:         2+ section-distinct sentences. Empty signatures are excluded.
  317:         Sentences within the same section that share a signature are
  318:         NOT counted as redundancy.
  319:     """
  320:     if section_order is None:
  321:         section_order = list(sections.keys())
  322: 
  323:     # Collect all locations with their signatures
  324:     all_locations: list[SentenceLocation] = []

  326:         if section_title not in sections:
  327:             continue
  328:         for idx, sentence in enumerate(sections[section_title]):
  329:             sig = extract_signature(sentence)
  330:             if sig.is_empty():
  331:                 continue

  333:                 section=section_title, index=idx,
  334:                 sentence=sentence, signature=sig,
  335:             ))

  337:     # Transitive clustering by compatibility (overlap + no-conflict):
  338:     # each location is compared against ALL members of each existing
  339:     # cluster. The candidate joins a cluster iff it has at least one
  340:     # overlapping member AND no conflicting member. When the candidate is
  341:     # compatible with multiple clusters, those clusters are merged into
  342:     # the lowest-index target ONLY IF the clusters themselves are
  343:     # pairwise-compatible (no cross-cluster member conflict). This
  344:     # prevents contextless-bridge merges (Codex brief-iter-1 P1 fix).
  345:     clusters: list[list[SentenceLocation]] = []
  346:     for loc in all_locations:
  347:         compat_idx = [
  348:             i for i, cluster in enumerate(clusters)
  349:             if _cluster_is_compatible(loc, cluster)
  350:         ]
  351:         if not compat_idx:
  352:             clusters.append([loc])
  353:             continue
  354:         target = clusters[compat_idx[0]]
  355:         target.append(loc)
  356:         # Codex brief-iter-2 P1 fix: check compatibility against the
  357:         # GROWING target (not the original), so two non-target clusters
  358:         # that conflict with each other cannot both be folded in via the
  359:         # same candidate bridge. Iteration order is the natural compat_idx
  360:         # order; for each follow-on candidate, the test is against the
  361:         # target's CURRENT membership.
  362:         merged_idx: list[int] = []
  363:         for j in compat_idx[1:]:
  364:             if _clusters_pairwise_compatible(target, clusters[j]):
  365:                 target.extend(clusters[j])
  366:                 merged_idx.append(j)
  367:         for j in sorted(merged_idx, reverse=True):
  368:             del clusters[j]
  369: 
  370:     groups: list[RedundancyGroup] = []
  371:     for cluster in clusters:
  372:         # Filter: at least 2 distinct sections must share this cluster
  373:         distinct_sections = {loc.section for loc in cluster}
  374:         if len(distinct_sections) < 2:
  375:             continue
  376:         # Within the cluster, primary = first location in section_order;
  377:         # redundants = all other locations (even same section as primary,
  378:         # if they appear after — but typically these will be from
  379:         # different sections).
  380:         primary = cluster[0]
  381:         redundants = cluster[1:]
  382:         groups.append(RedundancyGroup(
  383:             signature=primary.signature,
  384:             primary=primary,
  385:             redundants=redundants,
  386:         ))

  388:     return groups
