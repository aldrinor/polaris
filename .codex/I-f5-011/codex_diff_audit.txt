// xmur3 string-hash → 32-bit seed (public-domain implementation).
function xmur3(str: string): () => number {
  let h = 1779033703 ^ str.length;
  for (let i = 0; i < str.length; i++) {
    h = Math.imul(h ^ str.charCodeAt(i), 3432918353);
    h = (h << 13) | (h >>> 19);
  }
  return () => {
    h = Math.imul(h ^ (h >>> 16), 2246822507);
    h = Math.imul(h ^ (h >>> 13), 3266489909);
    h ^= h >>> 16;
    return h >>> 0;
  };
}

// sfc32 PRNG (public-domain) seeded from 4 32-bit ints.
function sfc32(a: number, b: number, c: number, d: number): () => number {
  return () => {
    a >>>= 0;
    b >>>= 0;
    c >>>= 0;
    d >>>= 0;
    let t = (a + b) | 0;
    a = b ^ (b >>> 9);
    b = (c + (c << 3)) | 0;
    c = (c << 21) | (c >>> 11);
    d = (d + 1) | 0;
    t = (t + d) | 0;
    c = (c + t) | 0;
    return (t >>> 0) / 4294967296;
  };
}

function pickIndices(): number[] {
  const seed_fn = xmur3(SEED);
  const rand = sfc32(seed_fn(), seed_fn(), seed_fn(), seed_fn());
  const all = Array.from({ length: N }, (_, i) => i);
  for (let i = N - 1; i > 0; i--) {
    const j = Math.floor(rand() * (i + 1));
    [all[i], all[j]] = [all[j], all[i]];
  }
  return all.slice(0, PICK_COUNT);
}

export function SentenceInspector({
  open,

  sentence,
  sentence_id,

// I-f5-003 — frontend mirror of polaris_graph.generator2.provenance.

const TOKEN_RE = /^\[#ev:([a-zA-Z0-9_\-]+):(\d+)-(\d+)\]$/;

export interface ParsedToken {
  raw: string;
  source_id: string;
  start: number;
  end: number;
}

export function parseProvenanceToken(token: string): ParsedToken | null {
  const match = TOKEN_RE.exec(token.trim());
  if (!match) return null;
  return {
    raw: token,
    source_id: match[1],
    start: parseInt(match[2], 10),
    end: parseInt(match[3], 10),
  };
}

export function parseAllTokens(tokens: string[]): ParsedToken[] {
  return tokens
    .map(parseProvenanceToken)
    .filter((t): t is ParsedToken => t !== null);
}

+function xmur3(str: string): () => number {
+  let h = 1779033703 ^ str.length;

+function sfc32(a: number, b: number, c: number, d: number): () => number {
+  return () => {

+    t = (t + d) | 0;
